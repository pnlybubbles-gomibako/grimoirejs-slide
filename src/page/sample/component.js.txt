gr.registerComponent("Rotate", {
  attributes: {
    speed: {
      default: 0.2,
      converter: "Number",
    },
  },
  $mount: function() {
    this.phi = 0;
  },
  $update: function() {
    this.phi += this.getAttribute("speed");
    this.node.setAttribute("rotation", "y(" + this.phi + ")");
  },
});
